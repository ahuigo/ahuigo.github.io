example

    var data = [];
    for(var i in list){
        var item = {x:moment(list[i].rettime) ,y:parseInt(list[i][key])}
        data.push(item);
    }
    var config = {
        type: 'line',
        data: {
            datasets: [{
                label: 'xtitle',
                data: data
            }]
        },
        options: {
            scales: {
                xAxes: [{
                    type: 'time',
                    position: 'bottom'
                }]
            }
        }
    };

    var $ctx = $('<canvas width="400" height="200"></canvas>');
    container.append($ctx);
    new Chart($ctx, config);
